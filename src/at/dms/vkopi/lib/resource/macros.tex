\catcode`\@ = 11
\def\StringEqual #1#2{%
    TT\fi
    \edef\@StringEqualOne{#1}%
    \edef\@StringEqualTwo{#2}%
    \ifx \@StringEqualOne\@StringEqualTwo
}
\def\EmptyString #1{%
    TT\fi
    \if\StringEqual{#1}{}%
}
\catcode`\@ = 12

\def\See #1{%
	\if\EmptyString{#1}%
	\else
	   siehe Absatz~\ref{#1} auf Seite~\pageref{#1}%
	\fi
}
